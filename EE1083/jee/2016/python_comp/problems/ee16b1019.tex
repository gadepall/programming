\subsection{For 2)}
\begin{lstlisting}[frame = single]
clear;
close;
clc;

a = -sqrt(2); b = 1 + sqrt(3);
x1 = linspace(0, 1, 1000);
x2 = linspace(1, sqrt(2), 1000);
x3 = linspace(sqrt(2), 5, 100000);

y1 = 2.*(x1.^2)./a;
y2 = a;
y3 = (2*(b^2) - 4*b)./(x3.^3);

plot(x1, y1, "2", x2, y2, "5", x3, y3, "3");
grid minor;
title("Problem 19\n 
     2.)  a = -sqrt(2); b = 1 + sqrt(3)\n
     f(x) is Discontinuous");
xlabel("x");
ylabel("f(x)");
\end{lstlisting}
